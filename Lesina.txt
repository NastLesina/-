#include <iostream>
#include <ctime>
using namespace std;
int main()
{
	setlocale(LC_ALL, "RUS");
	srand(time(NULL));
	int ROW = 10, COLUMN = 5;
	int **arr = new int*[ROW];
	int *result = new float[COLUMN]{0};
	for (int i = 0; i != ROW; i++)
		arr[i] = new int[COLUMN];
	for (int i = 0; i != ROW; i++)
	{
		for (int j = 0; j != COLUMN; j++)
		{
			arr[i][j] = rand() % 21 - 10;
			cout << arr[i][j] << "	";
		}
		cout << endl;
	}
	for (int i = 0; i != COLUMN; i++)
	{
		if (arr[i][0] > 0)
			result[0] += arr[i][0];
	}
	if (int i = 0; i != ROW; i++)
		for (int j = 0; j != COLUMN; j++)
		{
			while (arr[i][j] > 0)
				result[j] += arr[i][j];
		}
	for (int i = 0; i != COLUMN; i++)
		{
			result[i] = sqrt(result[i]);
			cout << "Рузультат =" << result[i] << endl;
		}
}
